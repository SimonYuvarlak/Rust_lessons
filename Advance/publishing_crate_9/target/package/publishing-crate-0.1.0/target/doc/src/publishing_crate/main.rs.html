/// Adds one to the given number
/// 
/// # Examples
/// 
/// ```
/// let num = 5;
/// let res = publishing_crate-module::add_one(num);
/// 
/// assert_eq!(6, res);
/// ```
pub fn add_one(num: i32) -> i32 {
    num + 1
}
